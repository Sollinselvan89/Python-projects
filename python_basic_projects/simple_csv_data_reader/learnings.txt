"""
Simple CSV Data Reader

This program reads a CSV file containing student records and allows the user
to display all records or apply simple filters (age > 15, grade == A).
It demonstrates file handling, CSV parsing, user input validation,
loop-based menus, and data cleaning.
"""

import csv


def get_user_input():
    """
    Prompt the user for a valid CSV file path and return it.

    Returns:
        str: The valid file path entered by the user.
    """
    while True:
        file_location = input("Enter the location of your file: ").strip()

        if not file_location:
            print("File location cannot be empty.")
            continue  # correction

        try:
            # correction: Try opening the file user provided, not students.csv
            with open(file_location, "r"):
                pass
            return file_location
        except FileNotFoundError:
            print("File not found! Please try again.")


def display_csv_pretty(filename):
    """
    Display all student records in a clean formatted style.

    Args:
        filename (str): Path to the CSV file.
    """
    # correction: Always open the file fresh to avoid pointer exhaustion
    with open(filename, "r") as file:
        reader = csv.DictReader(file)

        for row in reader:
            name = row.get("name", "").strip()
            age = row.get("age", "").strip()
            grade = row.get("grade", "").strip()

            print(f"Name: {name} | Age: {age} | Grade: {grade}")


def filter_age_above_15(filename):
    """
    Display all students whose age is greater than 15.

    Args:
        filename (str): Path to the CSV file.
    """
    with open(filename, "r") as file:  # correction
        reader = csv.DictReader(file)

        print("Students with age > 15")
        for row in reader:
            age = row.get("age", "").strip()

            if not age.isdigit():
                continue

            # correction: correct comparison and avoid >= 15 confusion
            if int(age) > 15:
                name = row.get("name", "").strip()
                grade = row.get("grade", "").strip()
                print(f"Name: {name} | Age: {age} | Grade: {grade}")


def filter_grade_A(filename):
    """
    Display all students whose grade is exactly 'A'.

    Args:
        filename (str): Path to the CSV file.
    """
    with open(filename, "r") as file:  # correction
        reader = csv.DictReader(file)

        print("Students with Grade == A")
        for row in reader:
            grade = row.get("grade", "").strip().upper()

            if grade == "A":
                name = row.get("name", "").strip()
                age = row.get("age", "").strip()
                print(f"Name: {name} | Age: {age} | Grade: {grade}")


def get_filter_choice():
    """
    Display filter options and return the user's choice.

    Returns:
        int: The user's selection (1‚Äì4).
    """
    print("\nChoose a filter:")
    print("1. Show all students")
    print("2. Age > 15")
    print("3. Grade == A")
    print("4. Quit")

    while True:
        choice = input("Enter your choice (1-4): ").strip()

        try:
            choice = int(choice)

            # correction: must allow 1‚Äì4, not 1‚Äì3
            if choice not in range(1, 5):
                print("Enter a value from 1 to 4.")
                continue
            return choice

        except ValueError:
            print("Please enter numbers only.")


def run_menu(filename):
    """
    Run the interactive menu allowing the user to choose actions.

    Args:
        filename (str): Path to the CSV file.
    """
    while True:
        choice = get_filter_choice()

        if choice == 1:
            display_csv_pretty(filename)
        elif choice == 2:
            filter_age_above_15(filename)
        elif choice == 3:
            filter_grade_A(filename)
        elif choice == 4:
            print("Goodbye!")
            break


def main():
    """
    Main program entry point.
    Collects file input and starts the menu loop.
    """
    filename = get_user_input()       # correction: get valid file path
    run_menu(filename)                # correction: pass filename, not file object


if __name__ == "__main__":
